import java.util.*;
class Example{
	public static int fact(int num){
		int f=1;
		for(int i=num;i>0;i--){
			f*=i;
		}
		return f;
	}
	public static void main(String args[]){
		Scanner input=new Scanner(System.in);
		System.out.print("Input a number : ");
		int num=input.nextInt();
		int f=fact(num);
		System.out.println(num+"! : "+f);
	}
}

	public static boolean binarySearch(int key,int[] x){
		int low=0;
		int upper=x.length;
		int mid=(low+upper)/2;
		if(key==x[mid]){
			return true;
		}else if(key>x[mid]){
			low=mid;
			mid=(low+x.length)/2;
			if(key==x[mid]){
				return true;
			}
		}else{
			upper=mid;
			mid=(upper+low)/2;
			if(key==x[mid]){
				return true;
			}
		}
		return false;	
	}
